/** * Calculates the total cost per linear foot for a sheet metal profile. * @param {number|string} stretchOutInches - The manually adjustable stretch out in inches. * @param {number|string} laborMinutesPerLF - The estimated labor time in minutes per linear foot. * @param {number|string} materialCostPerSqFt - Your cost of the flat sheet metal per square foot. * @param {number|string} hourlyLaborRate - Your shop/field hourly labor rate. * @param {number|string} wastePercentage - The percentage of material waste/scrap (e.g., 10 for 10%). * @returns {object} The broken down costs per linear foot as numbers. */ function calculateCostPerLF(stretchOutInches, laborMinutesPerLF, materialCostPerSqFt, hourlyLaborRate, wastePercentage = 0) { // Convert inputs to floats to prevent HTML string errors, defaulting to 0 if empty const stretch = parseFloat(stretchOutInches) || 0; const laborMins = parseFloat(laborMinutesPerLF) || 0; const materialCost = parseFloat(materialCostPerSqFt) || 0; const laborRate = parseFloat(hourlyLaborRate) || 0; const waste = parseFloat(wastePercentage) || 0; // 1. Calculate Material Cost per LF (with waste factor) const materialSqFtPerLF = stretch / 12; const baseMaterialCostPerLF = materialSqFtPerLF * materialCost; const wasteMultiplier = 1 + (waste / 100); const totalMaterialCostPerLF = baseMaterialCostPerLF * wasteMultiplier; // 2. Calculate Labor Cost per LF const laborRatePerMinute = laborRate / 60; const laborCostPerLF = laborMins * laborRatePerMinute; // 3. Total Cost per LF const totalCostPerLF = totalMaterialCostPerLF + laborCostPerLF; // Returns numbers so you can safely multiply them by the total job length later return { stretchOutUsed: stretch, materialCostLF: parseFloat(totalMaterialCostPerLF.toFixed(2)), laborCostLF: parseFloat(laborCostPerLF.toFixed(2)), totalCostLF: parseFloat(totalCostPerLF.toFixed(2)) }; } // EXAMPLE USAGE: // Coping Cap: 20" stretch, 5 mins labor, $1.50/sqft, $60/hr rate, 10% waste const copingEstimate = calculateCostPerLF(20, 5.0, 1.50, 60.00, 10); console.log(copingEstimate);
