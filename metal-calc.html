/** * Calculates the total cost per linear foot for a sheet metal profile. * * @param {number} stretchOutInches - The manually adjustable stretch out in inches. * @param {number} laborMinutesPerLF - The estimated labor time in minutes per linear foot. * @param {number} materialCostPerSqFt - Your cost of the flat sheet metal per square foot. * @param {number} hourlyLaborRate - Your shop/field hourly labor rate. * @returns {object} The broken down costs per linear foot. */ function calculateCostPerLF(stretchOutInches, laborMinutesPerLF, materialCostPerSqFt, hourlyLaborRate) { // 1. Calculate Material Cost per LF // Stretch out (inches) / 12 = sq ft of material used per linear foot of the profile const materialSqFtPerLF = stretchOutInches / 12; const materialCostPerLF = materialSqFtPerLF * materialCostPerSqFt; // 2. Calculate Labor Cost per LF const laborRatePerMinute = hourlyLaborRate / 60; const laborCostPerLF = laborMinutesPerLF * laborRatePerMinute; // 3. Total Cost per LF const totalCostPerLF = materialCostPerLF + laborCostPerLF; return { stretchOutUsed: stretchOutInches, materialCostLF: materialCostPerLF.toFixed(2), laborCostLF: laborCostPerLF.toFixed(2), totalCostLF: totalCostPerLF.toFixed(2) }; } // EXAMPLE USAGE: // Calculating a Coping Cap where you manually adjusted the stretch out to 20 inches. // Assuming metal costs $1.50/sq ft and shop labor is $60/hour. const copingEstimate = calculateCostPerLF(20, 5.0, 1.50, 60.00); console.log(copingEstimate); // Output: { stretchOutUsed: 20, materialCostLF: "2.50", laborCostLF: "5.00", totalCostLF: "7.50" }
